import { Tag } from 'lrc-ui';

# Tag 组件

### 基础标签

内置色彩，分为`default` `fill` `outline`三种类型标签。可使用`round`修改标签形状。

```tsx
function Demo() {
  const colors = [
    'blue',
    'red',
    'gray',
    'purple',
    'yellow',
    'green',
    'orange',
    'pink',
  ];
  return (
    <div>
      <style>{`
        .demo {
          display: flex;
          gap: 16px;
        }
      `}</style>
      <div className="demo">
        Default Tag：
        {colors.map((c) => (
          <Tag key={c} color={c}>
            {c}
          </Tag>
        ))}
      </div>
      <br />
      <div className="demo">
        Fill Tag：
        {colors.map((c) => (
          <Tag key={c} type="fill" color={c}>
            {c}
          </Tag>
        ))}
      </div>
      <br />
      <div className="demo">
        Outline Tag：
        {colors.map((c) => (
          <Tag key={c} type="outline" color={c}>
            {c}
          </Tag>
        ))}
      </div>
      <br />
      <div className="demo">
        Round Tag：
        {colors.map((c) => (
          <Tag key={c} round type="outline" color={c}>
            {c}
          </Tag>
        ))}
      </div>
    </div>
  );
}
```
