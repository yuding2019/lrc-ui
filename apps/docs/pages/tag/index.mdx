import { Tag } from 'lrc-ui';

# Tag 组件

### 基础标签

```tsx
function Demo() {
  const colors = ['blue', 'red', 'gray'];
  return (
    <div>
      <style>{`
        .demo {
          display: flex;
          gap: 16px;
        }
      `}</style>
      <div className="demo">
        {colors.map((c) => <Tag key={c} selectable color={c}>{c}</Tag>)}
      </div>
      <br />
      <div className="demo">
        {colors.map((c) => <Tag key={c} type="dark" color={c}>{c}</Tag>)}
      </div>
    </div>
  );
};
```
